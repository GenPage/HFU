setup_NCR_enclave = {
	# If the Nevada Enclave exists, transform it into the Western Enclave and
	# then join it.
	if = {
		limit = { nevada_enclave_exists = yes }
		var:GLOBAL.nevada_enclave_leader = {
			create_or_join_western_enclave = yes
			}
		every_country = {
			limit = {
				OR = {
					tag = ENC
					is_nevada_enclave_member = yes
				}
			}
			join_western_enclave = yes
		}
	}
	# Otherwise, create and join the Western Enclave by yourself Dayglow. :P
	else = {
		ENC = { create_or_join_western_enclave = yes }
	}

	set_politics = {
		ruling_party = elites
		elections_allowed = no
	}
	set_popularities = {
		people = 0
		elites = 100
		intellectuals = 0
		ruler = 0
	}
	hidden_effect = {
		set_technology = {
			power_armour_enclave_1 = 1
			energy_weaponry_tech_3 = 1
			energy_weaponry_tech_2 = 1
			energy_weaponry_tech_1 = 1
			vertibird_cas_tech = 1
			vertibird_tech = 1
			vertibird_unlock_tech = 1
			enclave_pa_unlock_tech = 1
			scientific_civilization = 1
		}
		load_oob = enclavecivilwar
		retire_country_leader = yes
		create_country_leader = {
			name = "Grant C. Hayes"
			desc = "Hayes_DESC"
			picture = "gfx/leaders/ENC/portrait_enc_hayes.dds"
			expire = "3000.1.1"
			ideology = militarist_technocracy

			traits = {

			}
		}
		set_country_flag = no_generic_advisor
		set_country_flag = no_generic_chief
		set_country_flag = no_generic_high_command
		set_country_flag = no_generic_theorist

		set_country_flag = give_settler_generic_political_advisors
		set_country_flag = give_settler_generic_chiefs
		set_country_flag = give_settler_generic_high_command
		set_country_flag = give_settler_generic_theorists
	}
	if = {
		limit = {
			NCR = {
				has_completed_focus = ncr_integration
			}
			BJS = {
				exists = yes
			}
		}
		NCR = {
			release_puppet = BJS
		}
		ENC = {
			puppet = BJS
		}
	}
}

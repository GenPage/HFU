transfer_twin_mothers_territory = {
	# Transfer every Twin Mothers core state owned by the spawner over to
	# the Twin Mothers.
	hidden_effect = {
		every_state = {
			limit = {
				is_core_of = TTM
				is_owned_by = PREV
			}
			TTM = {
				transfer_state = PREV
			}
		}
	}
}

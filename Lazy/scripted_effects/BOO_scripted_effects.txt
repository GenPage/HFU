transfer_boomer_territory = {
	# Transfer every Boomer core state owned by the spawner over to
	# the Boomers.
	hidden_effect = {
		every_state = {
			limit = {
				is_new_vegas_state = yes
				is_owned_by = PREV
			}
			BOO = {
				add_state_core = PREV
				transfer_state = PREV
			}
		}
	}
}

claim_missing_boomer_territory = {
	hidden_effect = {
		every_state = {
			limit = {
				is_new_vegas_state = yes
				NOT = { is_owned_by = BOO }
			}
			BOO = { add_state_claim = PREV }
		}
	}
}

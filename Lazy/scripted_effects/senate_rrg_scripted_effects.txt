	set_variable = { senate_party_b_value = 0 }
	set_variable = { senate_party_c_value = 0 }

	every_state = {
		limit = {
			is_RRG_states_at_height = yes
		}
		if = {
			limit = {
				check_variable = { senator_1_opinion_a = 3 }
			}
			add_to_variable = { PREV.senate_party_a_value = 1 }
		}
		if = {
			limit = {

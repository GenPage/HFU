defined_text = {
	name = GetPuppetLedgerManpower

	text = {
		trigger = {
			has_manpower < 1000
		}
		localization_key = "[?manpower|0]"
	}
	text = {
		trigger = {
			has_manpower < 1000000
		}
		localization_key = "[?manpower_k|2]K"
	}
	text = {
		trigger = {
			set_temp_variable = { manpower_temp = manpower_k }
			divide_temp_variable = { manpower_temp = 1000 }
			always = yes
		}
		localization_key = "[?manpower_temp|2]M"
	}
}

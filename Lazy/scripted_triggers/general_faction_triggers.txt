allowed_to_join_factions = {
	custom_trigger_tooltip = {
		tooltip = allowed_to_join_factions_tt
		is_subject = no
		is_in_faction = no
		NOT = { tag = TLA }
	}
}

not_allowed_to_join_factions = {
	custom_trigger_tooltip = {
		tooltip = not_allowed_to_join_factions_tt
		allowed_to_join_factions = no
	}
}

is_faction_of_one = {
	custom_trigger_tooltip = {
		tooltip = is_faction_of_one_tt
		is_faction_leader = yes
		num_faction_members < 2
	}
}

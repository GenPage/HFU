  NOT = { has_country_leader_with_trait = dislikes_ncr }
  has_government = people
  has_tech = settled_civilization
  is_not_raider_nation = yes
}

}

could_join_caesars_legion = {
  NOT = { has_country_leader_with_trait = dislikes_legion }

  # Countries that are settled or advanced will not be accepted into the legion.
  NOT = { has_tech = scientific_civilization }
  NOT = { has_tech = settled_civilization }

  # Countries run by people or intellectuals will also not try to join.
  NOT = {
    OR = {
      has_government = people
      has_government = intellectuals
    }
  }

  # You must also be human.
  is_human_nation = yes
}

could_join_brotherhood_of_steel = {
  NOT = { has_country_leader_with_trait = dislikes_brotherhood }
  OR = {
    has_tech = scientific_civilization
    AND = {
      has_tech = settled_civilization
      has_government = intellectuals
    }
  }
  is_human_nation = yes
}

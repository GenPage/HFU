is_valid_state_for_arms_factory = {
	free_building_slots = {
		building = arms_factory
		size > 0
		include_locked = yes
	}
	NOT = { has_state_flag = is_impassable }
	OR = {
		is_in_home_area = yes
		NOT = {
			owner = {
				any_owned_state = {
					free_building_slots = {
						building = arms_factory
						size > 0
						include_locked = yes
					}
					is_in_home_area = yes
				}
			}
		}
	}
}

can_ROOT_get_wargoal_on_THIS = {
	exists = yes
	NOT = { is_in_faction_with = ROOT }
	NOT = { is_subject_of = ROOT }
}

#Country scope

is_border_conflict_defender_vs_FROM = {
	has_variable = ROOT.defender_state_vs_@FROM
}

has_not_initiated_border_incident_with_FROM = {
	custom_trigger_tooltip = {
		tooltip = not_initiated_border_incident_with_FROM
		NOT = {
			any_state = {
				check_variable = { FROM.defender_state_vs_@PREV = id }
			}
		}
	}
}

}

#State scope
has_ROOT_at_least_1_div_in_current_state_scope = {
	custom_trigger_tooltip = {
		tooltip = at_least_one_division_in_state
		ROOT = { divisions_in_state = { state = PREV size > 0 } }
	}
}
#State scope
is_controlled_by_ROOT_or_subject = {
	CONTROLLER = {
		OR = {
			tag = ROOT
			is_subject_of = ROOT
		}
	}
}

#Country scope
is_free_or_subject_of_root = {
	OR = {
		is_subject = no
		is_subject_of = ROOT
	}
}

#Is Multiplayer
is_multiplayer = {
	any_country = {
		is_ai = no
		any_country = {
			NOT = {
				tag = PREV
			}
			is_ai = no
		}
	}
}

#use this to check if the debug global flag is set and the country is a player - useful to test stuff without having to tweak the code
is_in_debug_mode = {
	AND = {
		has_global_flag = debug
		is_ai = no
	}
}

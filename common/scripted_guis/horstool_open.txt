scripted_gui = {

	tool_open= {
		context_type = player_context

		window_name = "horsttool_open_container"
		visible = {
		has_country_flag = host
			NOT = { has_country_flag = hidden_switch }
		}

		effects = {
			tool_open_click = {
				if = {
					limit = { ROOT = { NOT = { has_country_flag = transfer_enabled } } }
					ROOT = { set_country_flag = transfer_enabled }
				}
				else = {
					ROOT = { clr_country_flag = transfer_enabled }
				}
			}
		}
		triggers = {
			tool_open_click_enabled = {
				always = yes
			}
		}
	}
	
	tool_close = {
		context_type = player_context

		window_name = "horsttool_close_container"

		visible = {
			always = yes
		}

		effects = {
			tool_close_click = {
				if = {
					limit = { NOT = { has_country_flag = hidden_switch } }
					set_country_flag = hidden_switch
				}
				else = {
					clr_country_flag = hidden_switch
				}
			}
		}
		triggers = {
			tool_close_click_enabled = {
				always = yes
			}
		}
	}

}
